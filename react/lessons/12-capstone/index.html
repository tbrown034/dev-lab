The Project: Campaign Finance Tracker

You're building a public-interest data journalism tool: a campaign finance tracker that lets citizens explore who's funding political campaigns in their state. Think ProPublica meets Next.js.

What you'll build

  • Public dashboard: Interactive D3 charts showing campaign contributions by sector, amount, and time
  • Search & filter: Real-time search across candidates and donors with React Query
  • Detail pages: Individual candidate profiles with contribution breakdowns
  • Admin panel: Authenticated editors can flag suspicious contributions and add notes
  • Data pipeline: Server Components process raw CSV data into visualizations

Why this project: It uses every pattern from the course: RSC for data processing, streaming for the dashboard, React Query for search, Server Actions for the admin panel, D3 for visualizations, Zod for validation, Testing Library for tests. It also reflects your journalism background — building tools that serve the public interest.

Step 2: Server Components for Data Processing

// lib/data-processing.ts (server-only — never shipped to client)
import { parse } from 'csv-parse/sync';
import { readFile } from 'fs/promises';

export async function processContributions() {
  const raw = await readFile('data/contributions.csv', 'utf-8');
  const records = parse(raw, { columns: true });

  // Heavy processing stays on the server
  const bySector = records.reduce((acc, r) => {
    const sector = r.contributor_sector;
    acc[sector] = (acc[sector] || 0) + parseFloat(r.amount);
    return acc;
  }, {} as Record<string, number>);

  const byMonth = records.reduce((acc, r) => {
    const month = r.date.substring(0, 7);
    acc[month] = (acc[month] || 0) + parseFloat(r.amount);
    return acc;
  }, {} as Record<string, number>);

  return { bySector, byMonth, total: records.length };
}

// app/dashboard/@charts/page.tsx (Server Component)
import { processContributions } from '@/lib/data-processing';
import { ContributionChart } from '@/components/charts/ContributionChart';

export default async function ChartsSlot() {
  // CSV parsing (csv-parse) and data crunching happen on server
  // Only the processed numbers go to the client
  const { bySector, byMonth } = await processContributions();

  return (
    <div className="charts-grid">
      <ContributionChart data={bySector} />  {/* Client: D3 rendering */}
      <TimelineChart data={byMonth} />       {/* Client: D3 rendering */}
    </div>
  );
}

The RSC payoff: The csv-parse library, all the data processing logic, and the raw CSV data stay on the server. The client receives only the aggregated numbers — maybe 2KB of JSON instead of 500KB of library code plus a 2MB CSV file. This is why RSC exists.

Step 3: D3 Visualizations as Client Components

// components/charts/ContributionChart.tsx
'use client';

import { useRef, useEffect } from 'react';
import * as d3 from 'd3';

interface Props {
  data: Record<string, number>;
}

export function ContributionChart({ data }: Props) {
  const svgRef = useRef<SVGSVGElement>(null);

  useEffect(() => {
    if (!svgRef.current) return;
    const svg = d3.select(svgRef.current);
    svg.selectAll('*').remove();

    const entries = Object.entries(data)
      .sort((a, b) => b[1] - a[1])
      .slice(0, 10);

    const margin = { top: 20, right: 30, bottom: 40, left: 120 };
    const width = 600 - margin.left - margin.right;
    const height = 400 - margin.top - margin.bottom;

    const g = svg
      .attr('viewBox', `0 0 600 400`)
      .append('g')
      .attr('transform', `translate(${margin.left},${margin.top})`);

    const x = d3.scaleLinear()
      .domain([0, d3.max(entries, d => d[1]) || 0])
      .range([0, width]);

    const y = d3.scaleBand()
      .domain(entries.map(d => d[0]))
      .range([0, height])
      .padding(0.2);

    g.selectAll('rect')
      .data(entries)
      .join('rect')
      .attr('x', 0)
      .attr('y', d => y(d[0]) || 0)
      .attr('height', y.bandwidth())
      .attr('fill', '#06b6d4')
      .attr('rx', 3)
      .transition()
      .duration(800)
      .attr('width', d => x(d[1]));

    g.selectAll('.label')
      .data(entries)
      .join('text')
      .attr('class', 'label')
      .attr('x', -5)
      .attr('y', d => (y(d[0]) || 0) + y.bandwidth() / 2)
      .attr('dy', '0.35em')
      .attr('text-anchor', 'end')
      .attr('fill', 'var(--text-dim)')
      .attr('font-size', '12px')
      .text(d => d[0]);

    g.append('g')
      .attr('transform', `translate(0,${height})`)
      .call(d3.axisBottom(x).ticks(5).tickFormat(d =>
        `$${(d as number / 1000000).toFixed(1)}M`
      ));

  }, [data]);

  return <svg ref={svgRef} className="w-full" />;
}

Step 4: Search with React Query

// hooks/useCandidateSearch.ts
'use client';

import { useQuery } from '@tanstack/react-query';
import { useDebounce } from './useDebounce';

export function useCandidateSearch(query: string) {
  const debouncedQuery = useDebounce(query, 300);

  return useQuery({
    queryKey: ['candidates', debouncedQuery],
    queryFn: async () => {
      if (!debouncedQuery) return [];
      const res = await fetch(
        `/api/candidates?q=${encodeURIComponent(debouncedQuery)}`
      );
      return res.json();
    },
    enabled: debouncedQuery.length > 0,
    staleTime: 60_000,
  });
}

// components/ui/CandidateSearch.tsx
'use client';

export function CandidateSearch() {
  const [query, setQuery] = useState('');
  const { data: candidates, isLoading, error } = useCandidateSearch(query);

  return (
    <div>
      <input
        role="searchbox"
        aria-label="Search candidates"
        value={query}
        onChange={e => setQuery(e.target.value)}
        placeholder="Search by name or office..."
      />

      {isLoading && <SearchSkeleton />}
      {error && <p>Search failed. Please try again.</p>}
      {candidates?.map(c => (
        <CandidateCard key={c.id} candidate={c} />
      ))}
    </div>
  );
}

Step 5: Admin Forms with Server Actions

// lib/schemas.ts — shared between client and server
import { z } from 'zod';

export const contributionNoteSchema = z.object({
  contributionId: z.string().uuid(),
  flagged: z.boolean(),
  note: z.string().max(1000).optional(),
  category: z.enum(['normal', 'suspicious', 'needs-review']),
});

export type ContributionNote = z.infer<typeof contributionNoteSchema>;

// app/actions/contributions.ts
'use server';

import { contributionNoteSchema } from '@/lib/schemas';
import { auth } from '@/lib/auth';

export async function flagContribution(prevState: any, formData: FormData) {
  // 1. Auth check
  const session = await auth();
  if (!session?.user?.isEditor) {
    return { error: 'Unauthorized' };
  }

  // 2. Validate with same Zod schema used on client
  const parsed = contributionNoteSchema.safeParse({
    contributionId: formData.get('contributionId'),
    flagged: formData.get('flagged') === 'true',
    note: formData.get('note'),
    category: formData.get('category'),
  });

  if (!parsed.success) {
    return { errors: parsed.error.flatten().fieldErrors };
  }

  // 3. Save to database
  await db.contributionNote.upsert({
    where: { contributionId: parsed.data.contributionId },
    create: parsed.data,
    update: parsed.data,
  });

  // 4. Revalidate
  revalidatePath(`/admin/contributions/${parsed.data.contributionId}`);
  return { success: true };
}

Step 7: Deployment Checklist

Before You Ship

  1. Performance: Run Lighthouse. Target 90+ on all scores. Check bundle size with ANALYZE=true.
  2. Accessibility: Tab through the entire app. Every interactive element should be keyboard-accessible. Run axe DevTools.
  3. Testing: All tests pass. Coverage on critical paths (search, filter, admin forms). At least one E2E test for the happy path.
  4. Error handling: Every data fetch has an error boundary. Server Actions return meaningful errors. 404 pages work.
  5. Security: Admin routes are auth-protected. Server Actions validate input. No sensitive data in client bundles.
  6. SEO: Metadata on all pages. Open Graph images. Proper heading hierarchy.
  7. Monitoring: Error tracking (Sentry). Analytics. Performance monitoring.
