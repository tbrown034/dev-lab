Sample Django View

By the end of this course, code like this will read like English.

# views.py — this is like a Next.js API route
from django.http import JsonResponse
from .models import Article

def latest_articles(request):
    # Like: const articles = await prisma.article.findMany()
    articles = Article.objects.filter(
        published=True
    ).order_by('-created_at')[:10]

    # Like: return NextResponse.json(data)
    data = [{
        'title': a.title,
        'author': a.author.name,
        'date': a.created_at.isoformat(),
    } for a in articles]

    return JsonResponse({'articles': data})
