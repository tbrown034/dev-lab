The Project: article-cli

You'll build article-cli — a command-line tool for managing a directory of Markdown articles. It reads, validates, transforms, and reports on article files. Think of it as an editorial toolkit for a static site.

# Usage examples
article-cli lint ./articles          # Validate all articles
article-cli stats ./articles         # Word counts, reading times, tag frequency
article-cli search "TypeScript"      # Full-text search across articles
article-cli convert ./articles --format json  # Convert MD to JSON
article-cli watch ./articles         # Watch for changes and auto-validate

Every concept from the course has a role here. Engine internals (understanding the event loop for the watch command). Types and coercion (robust input parsing). Closures (command factories). Prototypes and classes (command registry). Async (file I/O). Generators (streaming file processing). TypeScript (the entire type system). Testing (Vitest for every command). Node (streams, fs, path, process).

Step 2: Define the Type System

Start with types. If the types are right, the implementation flows naturally.

// src/types.ts

// ─── Branded types (lesson 09) ───
export type FilePath = string & { readonly __brand: "FilePath" };
export type Slug = string & { readonly __brand: "Slug" };

export function FilePath(path: string): FilePath {
  return path as FilePath;
}

export function Slug(slug: string): Slug {
  if (!/^[a-z0-9-]+$/.test(slug)) {
    throw new Error(`Invalid slug: ${slug}`);
  }
  return slug as Slug;
}

// ─── Article type (discriminated union for status) ───
export type ArticleFrontmatter = {
  title: string;
  slug: Slug;
  date: string;
  tags: string[];
  status: "draft" | "published" | "archived";
};

export type Article = {
  frontmatter: ArticleFrontmatter;
  body: string;
  filePath: FilePath;
  wordCount: number;
  readingTime: number;
};

// ─── Validation (discriminated union result) ───
export type ValidationResult =
  | { valid: true; article: Article }
  | { valid: false; errors: ValidationError[] };

export type ValidationError = {
  field: string;
  message: string;
  severity: "error" | "warning";
};

// ─── Command system (generics) ───
export interface Command<TArgs = void> {
  name: string;
  description: string;
  run(args: TArgs): Promise<CommandResult>;
}

export type CommandResult =
  | { success: true; message: string }
  | { success: false; error: string; exitCode: number };

// ─── CLI arguments (mapped type for type-safe parsing) ───
export type CLIArgs = {
  command: string;
  path?: string;
  flags: Record<string, string | boolean>;
};

Notice the course concepts: Branded types prevent mixing up file paths and slugs. Discriminated unions make invalid states unrepresentable. Generics make the command interface reusable. Mapped types handle flag parsing. This is TypeScript at its best — catching entire categories of bugs at compile time.

Step 3: Build the Markdown Parser

// src/utils/parser.ts
import { Article, ArticleFrontmatter, FilePath } from '../types.js';

export function parseArticle(content: string, filePath: FilePath): Article {
  const fmMatch = content.match(/^---\n([\s\S]*?)\n---\n([\s\S]*)$/);

  if (!fmMatch) {
    throw new Error(`No frontmatter found in ${filePath}`);
  }

  const [, fmRaw, body] = fmMatch;
  const frontmatter = parseFrontmatter(fmRaw);
  const wordCount = body.trim().split(/\s+/).filter(Boolean).length;

  return {
    frontmatter,
    body: body.trim(),
    filePath,
    wordCount,
    readingTime: Math.ceil(wordCount / 200),
  };
}

function parseFrontmatter(raw: string): ArticleFrontmatter {
  const lines = raw.split('\n');
  const result: Record<string, unknown> = {};

  for (const line of lines) {
    const [key, ...rest] = line.split(':');
    if (!key) continue;
    const value = rest.join(':').trim();

    if (value.startsWith('[')) {
      // Simple array parsing: [tag1, tag2, tag3]
      result[key.trim()] = value.slice(1, -1).split(',').map(s => s.trim());
    } else {
      result[key.trim()] = value;
    }
  }

  return result as ArticleFrontmatter;
}

Step 5: Implement Commands

The Lint Command

// src/commands/lint.ts
import { readdir, readFile } from 'fs/promises';
import { join } from 'path';
import { Command, CommandResult, FilePath, ValidationResult } from '../types.js';
import { parseArticle } from '../utils/parser.js';
import { validateArticle } from '../utils/validator.js';

export const lintCommand: Command<{ path: string }> = {
  name: 'lint',
  description: 'Validate all Markdown articles in a directory',

  async run({ path }): Promise<CommandResult> {
    const files = await readdir(path);
    const mdFiles = files.filter(f => f.endsWith('.md'));

    if (mdFiles.length === 0) {
      return { success: false, error: 'No .md files found', exitCode: 1 };
    }

    const results: Array<{ file: string; result: ValidationResult }> = [];

    // Process files in parallel (lesson 05: Promise.all)
    const validations = await Promise.all(
      mdFiles.map(async (file) => {
        const filePath = FilePath(join(path, file));
        const content = await readFile(filePath, 'utf-8');
        const article = parseArticle(content, filePath);
        const result = validateArticle(article);
        return { file, result };
      })
    );

    const errors = validations.filter(v => !v.result.valid);

    if (errors.length === 0) {
      return {
        success: true,
        message: `All ${mdFiles.length} articles passed validation.`
      };
    }

    // Format error report
    const report = errors.map(({ file, result }) => {
      if (!result.valid) {
        const msgs = result.errors.map(e => `  ${e.severity}: ${e.field} — ${e.message}`);
        return `${file}:\n${msgs.join('\n')}`;
      }
    }).join('\n\n');

    return {
      success: false,
      error: `${errors.length} articles have issues:\n\n${report}`,
      exitCode: 1
    };
  }
};

The Stats Command (using generators)

// src/commands/stats.ts — uses generators for lazy processing
import { readdir, readFile } from 'fs/promises';
import { join } from 'path';
import { Command, CommandResult, FilePath, Article } from '../types.js';
import { parseArticle } from '../utils/parser.js';

// Async generator: process articles lazily (lesson 06)
async function* loadArticles(dirPath: string): AsyncGenerator<Article> {
  const files = await readdir(dirPath);

  for (const file of files) {
    if (!file.endsWith('.md')) continue;
    const filePath = FilePath(join(dirPath, file));
    const content = await readFile(filePath, 'utf-8');
    yield parseArticle(content, filePath);
  }
}

export const statsCommand: Command<{ path: string }> = {
  name: 'stats',
  description: 'Generate statistics for all articles',

  async run({ path }): Promise<CommandResult> {
    let totalWords = 0;
    let totalArticles = 0;
    const tagCounts = new Map<string, number>();
    const statusCounts = new Map<string, number>();

    for await (const article of loadArticles(path)) {
      totalArticles++;
      totalWords += article.wordCount;

      for (const tag of article.frontmatter.tags) {
        tagCounts.set(tag, (tagCounts.get(tag) || 0) + 1);
      }

      const status = article.frontmatter.status;
      statusCounts.set(status, (statusCounts.get(status) || 0) + 1);
    }

    const avgWords = totalArticles > 0 ? Math.round(totalWords / totalArticles) : 0;

    const topTags = [...tagCounts.entries()]
      .sort((a, b) => b[1] - a[1])
      .slice(0, 5)
      .map(([tag, count]) => `${tag} (${count})`)
      .join(', ');

    const message = [
      `Articles: ${totalArticles}`,
      `Total words: ${totalWords.toLocaleString()}`,
      `Average words: ${avgWords}`,
      `Average reading time: ${Math.ceil(avgWords / 200)} min`,
      `Top tags: ${topTags}`,
      `By status: ${[...statusCounts.entries()].map(([s, c]) => `${s}: ${c}`).join(', ')}`,
    ].join('\n');

    return { success: true, message };
  }
};

Step 6: CLI Entry Point

// src/index.ts
#!/usr/bin/env node

import { argv, exit } from 'process';
import { lintCommand } from './commands/lint.js';
import { statsCommand } from './commands/stats.js';
import { Command, CLIArgs } from './types.js';

// Type-safe command registry (lesson 08: generics)
const commands = new Map<string, Command<any>>([
  ['lint', lintCommand],
  ['stats', statsCommand],
]);

function parseArgs(rawArgs: string[]): CLIArgs {
  const [command = 'help', path, ...rest] = rawArgs;
  const flags: Record<string, string | boolean> = {};

  for (const arg of rest) {
    if (arg.startsWith('--')) {
      const [key, value] = arg.slice(2).split('=');
      flags[key] = value ?? true;
    }
  }

  return { command, path, flags };
}

async function main() {
  const args = parseArgs(argv.slice(2));
  const command = commands.get(args.command);

  if (!command) {
    console.error(`Unknown command: ${args.command}`);
    console.log(`Available commands: ${[...commands.keys()].join(', ')}`);
    exit(1);
  }

  if (!args.path) {
    console.error(`Usage: article-cli ${args.command} <path>`);
    exit(1);
  }

  const result = await command.run({ path: args.path, flags: args.flags });

  if (result.success) {
    console.log(result.message);
  } else {
    console.error(result.error);
    exit(result.exitCode);
  }
}

main().catch((err) => {
  console.error('Unexpected error:', err);
  exit(1);
});

  1. Watch mode — Use fs.watch() to monitor the articles directory and auto-lint on changes. This exercises the event loop and callback patterns.
