01 / PERFORMANCE

Debounce & Throttle

Search inputs, scroll handlers, resize events
function debounce<T extends (...args: any[]) => void>(
  fn: T, delay: number
): (...args: Parameters<T>) => void {
  let timer: ReturnType<typeof setTimeout>;
  return (...args) => {
    clearTimeout(timer);
    timer = setTimeout(() => fn(...args), delay);
  };
}

// Usage: auto-save, search-as-you-type
const search = debounce(query => {
  fetchResults(query);
}, 300);
  • Debounce: waits until input stops for N ms before firing
  • Throttle: fires at most once per N ms during continuous input
  • Type-safe with generics preserving parameter types

02 / ARCHITECTURE

Pub/Sub Event System

Decoupled component communication
type EventMap = Record<string, any>;

class EventBus<T extends EventMap> {
  private handlers = new Map<
    keyof T, Set<Function>
  >();

  on<K extends keyof T>(
    event: K, handler: (data: T[K]) => void
  ) {
    if (!this.handlers.has(event))
      this.handlers.set(event, new Set());
    this.handlers.get(event)!.add(handler);
    return () => this.handlers.get(event)!.delete(handler);
  }

  emit<K extends keyof T>(event: K, data: T[K]) {
    this.handlers.get(event)?.forEach(h => h(data));
  }
}
  • Fully type-safe: event names and payloads are checked at compile time
  • Returns an unsubscribe function for cleanup
  • Foundation for micro-frontend and plugin architectures
EventEmitter Generics Type-safe
03 / STATE

State Machine

Forms, workflows, UI interactions
type State = "idle" | "loading" | "success" | "error";
type Event = "FETCH" | "RESOLVE" | "REJECT" | "RETRY";

const transitions: Record<State,
  Partial<Record<Event, State>>
> = {
  idle:    { FETCH: "loading" },
  loading: { RESOLVE: "success", REJECT: "error" },
  success: { FETCH: "loading" },
  error:   { RETRY: "loading" },
};

function transition(state: State, event: Event): State {
  return transitions[state][event] ?? state;
}
  • Eliminates impossible states with discriminated unions
  • Transitions are a pure function: easy to test and reason about
  • Scales to complex workflows with guard conditions and side effects
Discriminated Unions Pure Functions XState-pattern
04 / RESILIENCE

Retry with Exponential Backoff

API calls, network requests, flaky services
async function retry<T>(
  fn: () => Promise<T>,
  opts = { retries: 3, baseDelay: 1000 }
): Promise<T> {
  for (let attempt = 0; attempt <= opts.retries; attempt++) {
    try {
      return await fn();
    } catch (err) {
      if (attempt === opts.retries) throw err;
      const delay = opts.baseDelay * 2 ** attempt;
      const jitter = delay * (0.5 + Math.random() * 0.5);
      await new Promise(r => setTimeout(r, jitter));
    }
  }
  throw new Error("unreachable");
}
  • Exponential backoff: 1s, 2s, 4s... prevents server overload
  • Jitter randomization prevents thundering herd problem
  • Generic return type preserves the wrapped function's type
async/await Generics Error Handling
05 / CONCURRENCY

Worker Thread Pool

CPU-intensive tasks, image processing, parsing
class WorkerPool {
  private workers: Worker[] = [];
  private queue: Array<{
    data: any;
    resolve: Function;
    reject: Function;
  }> = [];
  private available: Set<Worker>;

  constructor(script: string, size = navigator.hardwareConcurrency) {
    this.workers = Array.from({ length: size },
      () => new Worker(script, { type: "module" })
    );
    this.available = new Set(this.workers);
  }

  exec<T>(data: any): Promise<T> {
    return new Promise((resolve, reject) => {
      this.queue.push({ data, resolve, reject });
      this.dispatch();
    });
  }
}
  • Manages a fixed pool of Web Workers for parallel execution
  • Queue-based dispatch prevents thread oversubscription
  • Uses navigator.hardwareConcurrency for optimal pool sizing

06 / DATA

Stream Processing

Large files, real-time feeds, ETL pipelines
async function* processStream<T, R>(
  source: AsyncIterable<T>,
  transform: (chunk: T) => R,
  filter?: (item: R) => boolean
): AsyncGenerator<R> {
  for await (const chunk of source) {
    const result = transform(chunk);
    if (!filter || filter(result)) {
      yield result;
    }
  }
}

// Usage: process a large NDJSON file
for await (const record of processStream(
  readLines("data.ndjson"),
  line => JSON.parse(line),
  obj => obj.score > 0.8
)) { /* ... */ }
  • Async generators enable lazy, memory-efficient data processing
  • Composable transform + filter pipeline pattern
  • Handles backpressure naturally through async iteration

Type-safe API Client

REST APIs, microservices, BFFs
interface ApiRoutes {
  "/users":     { GET: User[];   POST: User };
  "/users/:id": { GET: User;     PUT: User };
  "/posts":     { GET: Post[];   POST: Post };
}

class ApiClient<R extends Record<string, Record<string, any>>> {
  async request<
    P extends keyof R & string,
    M extends keyof R[P] & string
  >(method: M, path: P): Promise<R[P][M]> {
    const res = await fetch(path, { method });
    return res.json();
  }
}

// TypeScript knows this returns User[]
const users = await api.request("GET", "/users");
  • Route definitions map paths to method/response-type pairs
  • Compiler catches invalid path + method combinations
  • Response types are automatically inferred -- no casting needed

08 / ARCHITECTURE

Plugin Architecture

Build tools, editors, frameworks
interface Plugin {
  name: string;
  version: string;
  setup(ctx: PluginContext): void | Promise<void>;
  teardown?(): void;
}

class PluginManager {
  private plugins = new Map<string, Plugin>();
  private hooks = new Map<string, Set<Function>>();

  async register(plugin: Plugin) {
    const ctx: PluginContext = {
      hook: (name, fn) => {
        if (!this.hooks.has(name))
          this.hooks.set(name, new Set());
        this.hooks.get(name)!.add(fn);
      }
    };
    await plugin.setup(ctx);
    this.plugins.set(plugin.name, plugin);
  }
}
  • Plugins register hooks during setup phase via a context object
  • Lifecycle-based: setup / teardown for clean resource management
  • Same pattern used by Vite, Rollup, and Webpack

09 / REACTIVE

Observable Pattern

RxJS-style streams, reactive state
class Observable<T> {
  constructor(
    private subscribeFn: (
      observer: Observer<T>
    ) => Teardown
  ) {}

  subscribe(observer: Partial<Observer<T>>): Subscription {
    const teardown = this.subscribeFn({
      next: observer.next ?? (() => {}),
      error: observer.error ?? (() => {}),
      complete: observer.complete ?? (() => {}),
    });
    return { unsubscribe: teardown };
  }

  pipe<R>(op: (src: Observable<T>) => Observable<R>) {
    return op(this);
  }
}
  • Push-based data flow with next/error/complete lifecycle
  • Composable via pipe() with map, filter, debounce operators
  • Unsubscribe function prevents memory leaks
Observer Reactive Pipe
10 / TESTING

Dependency Injection

Testable services, decoupled modules
const tokens = {
  logger: Symbol("Logger"),
  db:     Symbol("Database"),
} as const;

class Container {
  private registry = new Map<symbol, () => any>();

  bind<T>(token: symbol, factory: () => T) {
    this.registry.set(token, factory);
  }

  get<T>(token: symbol): T {
    const factory = this.registry.get(token);
    if (!factory) throw new Error(`No binding: ${token.toString()}`);
    return factory() as T;
  }
}

// Test: swap real DB for mock
container.bind(tokens.db, () => new MockDB());
  • Symbol-based tokens prevent naming collisions
  • Factory functions enable lazy instantiation and scoping
  • Swap real implementations for mocks in tests without code changes

11 / UX

Command Pattern (Undo/Redo)

Text editors, drawing apps, form builders
interface Command {
  execute(): void;
  undo(): void;
  description: string;
}

class CommandHistory {
  private undoStack: Command[] = [];
  private redoStack: Command[] = [];

  execute(cmd: Command) {
    cmd.execute();
    this.undoStack.push(cmd);
    this.redoStack.length = 0; // clear redo
  }

  undo() {
    const cmd = this.undoStack.pop();
    if (!cmd) return;
    cmd.undo();
    this.redoStack.push(cmd);
  }

  redo() {
    const cmd = this.redoStack.pop();
    if (!cmd) return;
    cmd.execute();
    this.undoStack.push(cmd);
  }
}
  • Each action is an object with execute() and undo() methods
  • Redo stack clears on new command (standard undo/redo semantics)
  • Description field enables "Undo: Delete paragraph" UI labels

Memoization with Cache Eviction

Expensive computations, API responses
function memoize<Args extends any[], R>(
  fn: (...args: Args) => R,
  opts: { maxSize?: number; ttl?: number } = {}
) {
  const cache = new Map<string, {
    value: R; expires: number
  }>();

  return (...args: Args): R => {
    const key = JSON.stringify(args);
    const hit = cache.get(key);
    if (hit && (!opts.ttl || Date.now() < hit.expires))
      return hit.value;

    const result = fn(...args);
    if (opts.maxSize && cache.size >= opts.maxSize)
      cache.delete(cache.keys().next().value!);
    cache.set(key, {
      value: result,
      expires: Date.now() + (opts.ttl ?? Infinity)
    });
    return result;
  };
}
  • LRU-style eviction when maxSize is reached (Map preserves insertion order)
  • Optional TTL for time-based cache invalidation
  • Preserves function signature via generic types
Map Caching Generics
13 / RESILIENCE

Circuit Breaker

Microservices, API gateways, external integrations
class CircuitBreaker {
  private state: "closed" | "open" | "half-open" = "closed";
  private failures = 0;
  private lastFailure = 0;

  constructor(
    private threshold = 5,
    private resetTimeout = 30000
  ) {}

  async call<T>(fn: () => Promise<T>): Promise<T> {
    if (this.state === "open") {
      if (Date.now() - this.lastFailure > this.resetTimeout)
        this.state = "half-open";
      else throw new Error("Circuit is OPEN");
    }
    try {
      const result = await fn();
      this.onSuccess();
      return result;
    } catch (err) {
      this.onFailure();
      throw err;
    }
  }
}
  • Three states: closed (normal), open (failing fast), half-open (testing)
  • Prevents cascading failures across distributed services
  • Auto-recovers after resetTimeout with a test request

14 / SECURITY

Rate Limiter (Token Bucket)

API endpoints, WebSocket messages, form submissions
class RateLimiter {
  private tokens: number;
  private lastRefill: number;

  constructor(
    private maxTokens: number,
    private refillRate: number // tokens per second
  ) {
    this.tokens = maxTokens;
    this.lastRefill = Date.now();
  }

  tryConsume(count = 1): boolean {
    this.refill();
    if (this.tokens >= count) {
      this.tokens -= count;
      return true;
    }
    return false;
  }

  private refill() {
    const now = Date.now();
    const elapsed = (now - this.lastRefill) / 1000;
    this.tokens = Math.min(
      this.maxTokens,
      this.tokens + elapsed * this.refillRate
    );
    this.lastRefill = now;
  }
}
  • Token bucket algorithm: smooth rate limiting with burst allowance
  • Tokens refill continuously based on elapsed time
  • Used in Express middleware, WebSocket handlers, and client-side throttling

Builder Pattern (Fluent API)

Query builders, config objects, test fixtures
class QueryBuilder<T> {
  private conditions: string[] = [];
  private sortField?: keyof T;
  private limitVal?: number;

  where<K extends keyof T>(
    field: K, op: "=" | ">" | "<", val: T[K]
  ): this {
    this.conditions.push(`${String(field)} ${op} ${val}`);
    return this;
  }

  orderBy(field: keyof T): this {
    this.sortField = field;
    return this;
  }

  limit(n: number): this { this.limitVal = n; return this; }

  build(): string { /* ... */ }
}

// Fluent, type-safe chaining
new QueryBuilder<User>()
  .where("age", ">", 18)
  .orderBy("name")
  .limit(10)
  .build();
  • return this enables fluent method chaining
  • Type-safe field names: compiler catches typos in "where" clauses
  • Same pattern used by Knex, Prisma, and Zod
Fluent API keyof Method Chaining
